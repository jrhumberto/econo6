def create_project_structure():
    """Cria toda a estrutura do projeto recursivamente"""
    print("üöÄ Iniciando gera√ß√£o do projeto...")
    
    # Criar diret√≥rios base
    base_dirs = set()
    for file_path in PROJECT_FILES.keys():
        dir_path = os.path.dirname(file_path)
        if dir_path:
            base_dirs.add(dir_path)
    
    # Criar todos os diret√≥rios recursivamente
    for dir_path in sorted(base_dirs):
        os.makedirs(dir_path, exist_ok=True)
        print(f"üìÅ Diret√≥rio criado: {dir_path}")
    
    # Criar todos os arquivos
    for file_path, content in PROJECT_FILES.items():
        with open(file_path, 'w', encoding='utf-8') as f:
            f.write(content)
        print(f"‚úÖ Arquivo criado: {file_path}")
    
    print("\n‚ú® Projeto gerado com sucesso!")
    print("\nüì¶ Estrutura criada:")
    print_tree(".", prefix="")

def print_tree(directory, prefix=""):
    """Imprime a √°rvore de diret√≥rios"""
    try:
        entries = sorted(os.listdir(directory))
        dirs = [e for e in entries if os.path.isdir(os.path.join(directory, e)) and not e.startswith('.')]
        files = [e for e in entries if os.path.isfile(os.path.join(directory, e))]
        
        for d in dirs:
            print(f"{prefix}‚îú‚îÄ‚îÄ {d}/")
            print_tree(os.path.join(directory, d), prefix + "‚îÇ   ")
        
        for f in files:
            print(f"{prefix}‚îú‚îÄ‚îÄ {f}")
    except PermissionError:
        pass

if __name__ == "__main__":
    create_project_structure()
